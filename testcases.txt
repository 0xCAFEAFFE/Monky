#include <criterion/criterion.h>

#define TRUE -1
#define FALSE 0

#define ERROR_INPUT_EMPTY       -1
#define ERROR_STACK_UNDERFLOW   -2
#define ERROR_UNKNOWN_CMD       -3
#define ERROR_LITERAL_INVALID   -4
#define ERROR_BLOCK_END         -5
#define ERROR_LOOP_START        -6
#define ERROR_STRING_END        -7
#define ERROR_LITERAL_SIZE      -8
#define ERROR_DATA_INDEX        -9
#define ERROR_STACK_OVERFLOW    -10
#define ERROR_SUBROUTINE_UNDEF  -11

char monky(const char*);

Test(sample_tests, should_pass_all_the_tests_provided)
{
    // errors
    cr_assert_eq(monky(""), ERROR_INPUT_EMPTY);
    cr_assert_eq(monky(" "), ERROR_INPUT_EMPTY);
    cr_assert_eq(monky("{"), ERROR_UNKNOWN_CMD);
    cr_assert_eq(monky("1a"), ERROR_LITERAL_INVALID);
    cr_assert_eq(monky("a1"), ERROR_LITERAL_INVALID);
    cr_assert_eq(monky("."), ERROR_STACK_UNDERFLOW);
    cr_assert_eq(monky("%"), ERROR_STACK_UNDERFLOW);
    cr_assert_eq(monky("_"), ERROR_STACK_UNDERFLOW);
    cr_assert_eq(monky("\"hello"), ERROR_STRING_END);
    cr_assert_eq(monky("234"), ERROR_LITERAL_SIZE);
    cr_assert_eq(monky("-234"), ERROR_LITERAL_SIZE);
    cr_assert_eq(monky("1 123 :"), ERROR_DATA_INDEX);
    cr_assert_eq(monky("1 94 :"), ERROR_DATA_INDEX);
    cr_assert_eq(monky("1 2 3 4 5 5 \\"), ERROR_STACK_UNDERFLOW);
    cr_assert_eq(monky("1 ( 2 3"), ERROR_BLOCK_END);
    cr_assert_eq(monky("1 2 ] 3"), ERROR_LOOP_START);
    cr_assert_eq(monky("1 -1 \\"), ERROR_STACK_OVERFLOW);
    cr_assert_eq(monky("A ;"), ERROR_SUBROUTINE_UNDEF);
  
    // stack
    cr_assert_eq(monky("1 ."), 1);
    cr_assert_eq(monky("a"), 'a');
    cr_assert_eq(monky("B"), 'B');
    cr_assert_eq(monky("1 2 ."), 2);
    cr_assert_eq(monky("1 a ,"), 1); 
    cr_assert_eq(monky("2 3 _"), 2);
    cr_assert_eq(monky("1 % +"), 2);
    cr_assert_eq(monky("1 2 $"), 1);
    cr_assert_eq(monky("9 5 2 - -"), 6);
    cr_assert_eq(monky("2 9 5 @ - -"), 6);
    cr_assert_eq(monky("1 2 3 ^"), 2);
    cr_assert_eq(monky("1 2 3 4 5 0 \\"), 5);
    cr_assert_eq(monky("1 2 3 4 5 2 \\"), 3);
    cr_assert_eq(monky("1 2 3 4 5 4 \\"), 1);
  
    // math
    cr_assert_eq(monky("1"), 1);
    cr_assert_eq(monky("1 2"), 2);
    cr_assert_eq(monky("1 2 +"), 3);
    cr_assert_eq(monky("1 -2 +"), -1);
    cr_assert_eq(monky("3 2 -"), 1);
    cr_assert_eq(monky("-3 -2 -"), -1);
    cr_assert_eq(monky("3 2 *"), 6);
    cr_assert_eq(monky("-3 2 *"), -6);
    cr_assert_eq(monky("-3 -2 *"), 6);
    cr_assert_eq(monky("8 2 /"), 4);
    cr_assert_eq(monky("-8 2 /"), -4);
    cr_assert_eq(monky("8 -2 /"), -4);
    cr_assert_eq(monky("12 % 5 / 5 * -"), 2); // 12 mod 5
  
    // logic
    cr_assert_eq(monky("12 10 &"), 8);
    cr_assert_eq(monky("12 10 |"), 14);
    cr_assert_eq(monky("12 10 #"), 6);
    cr_assert_eq(monky("127 ~"), -128);
    cr_assert_eq(monky("-1 ~"), 0);
    cr_assert_eq(monky("0 ~"), -1);
  
    // comparison
    cr_assert_eq(monky("1 2 = "), FALSE);
    cr_assert_eq(monky("3 3 = "), TRUE);
    cr_assert_eq(monky("-3 -3 = "), TRUE);
    cr_assert_eq(monky("1 1 < "), FALSE);
    cr_assert_eq(monky("1 2 < "), FALSE);
    cr_assert_eq(monky("2 1 < "), TRUE);
    cr_assert_eq(monky("1 2 > "), TRUE);
    cr_assert_eq(monky("2 1 > "), FALSE);
    cr_assert_eq(monky("2 2 > "), FALSE);
    cr_assert_eq(monky("-1 -2 < "), TRUE);
    cr_assert_eq(monky("-1 -1 < "), FALSE);
    cr_assert_eq(monky("-2 -1 > "), TRUE);
    cr_assert_eq(monky("-1 -2 > "), FALSE);
    cr_assert_eq(monky("-1 1 > "), TRUE);
    cr_assert_eq(monky("-1 1 < "), FALSE);
    cr_assert_eq(monky("1 -1 > "), FALSE);
    cr_assert_eq(monky("1 -1 < "), TRUE);

    // variables
    cr_assert_eq(monky("2 a :"), 2);
    cr_assert_eq(monky("2 4 a : 5 a ;"), 4);
    cr_assert_eq(monky("2 4 a : _ b : _ a ;"), 4);
    cr_assert_eq(monky("2 4 a : _ b : _ b ;"), 2);
    cr_assert_eq(monky("0 4 -100 : _"), 0);
    cr_assert_eq(monky("3 -100 : _ 2 -100 ;"), 3);
    cr_assert_eq(monky("2 -123 : _ 3 -123 ;"), 2);
    cr_assert_eq(monky("0 5 -100 : _ -99 1 - ;"), 5);
 
    // conditions
    cr_assert_eq(monky("1 0 ? 2 +"), 2);
    cr_assert_eq(monky("0 3 ? 2 +"), 3);
    cr_assert_eq(monky("1 0 ! 2 +"), 1);
    cr_assert_eq(monky("1 2 ! 3 +"), 5);

    // blocks
    cr_assert_eq(monky("1 ( 2 3 4 )"), 1);
    cr_assert_eq(monky("1 ? ( 4 )"), 4);
    cr_assert_eq(monky("0 ? ( 2 3 5 )"), 0);
    cr_assert_eq(monky("2 ! ( 2 3 4 )"), 2);
    cr_assert_eq(monky("0 ! ( 2 3 6 )"), 6);

    // loops
    cr_assert_eq(monky("3 [ 1 - ! ]"), 0);
    cr_assert_eq(monky("0 o l l e h 10 [ , ! ]"), 0);

    // strings
    cr_assert_eq(monky("\"hellorld\""), 'h');
    cr_assert_eq(monky("0 ( \"comment\" )"), 0);
    cr_assert_eq(monky("1 2 3 \"hello\""), 'h');
    cr_assert_eq(monky("\"wello horld\""), 'w');
    cr_assert_eq(monky("\"test\" 1"), 1);
    cr_assert_eq(monky("\"test\" 1 2 +"), 3);
    cr_assert_eq(monky("1 \"\nnew\nline\n\" [ , ! ] _"), 1);

    // move string from stack to array
    cr_assert_eq(monky("\"hello\" -128 [ $ ^ : ? ( _ 1 + ] )"), 0);

    // subroutines
    cr_assert_eq(monky("1 2 ( 3 + ) M :"), 2);
    cr_assert_eq(monky("1 2 ( 3 + ) B : 4 B ; 5 ^"), 7);

    // what about loops in functions ?
    
    // not working yet
    //cr_assert_eq(monky("1 ( 2 ( ) 3 4  )"), 1);
}
